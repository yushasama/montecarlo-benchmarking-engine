73inline int countInsideCircle_AVX(__m256d x, __m256d y) {
74 __m256d x2 = _mm256_mul_pd(x, x);
75 __m256d y2 = _mm256_mul_pd(y, y);
76 __m256d dist2 = _mm256_add_pd(x2, y2);
77 __m256d ones = _mm256_set1_pd(1.0);
78 __m256d cmp = _mm256_cmp_pd(dist2, ones, _CMP_LE_OQ);
79 return __builtin_popcount(_mm256_movemask_pd(cmp));
80}
81#endif

171inline int* monteCarloPI_SIMD(int numberOfTrials) {
172 thread_local PoolAllocator pool(64 * 1024);
173 pool.reset();
174
175 int* hits = pool.allocate<int>();
176 if (!hits) {
177 std::cerr << "[ERROR] PoolAllocator ran out of memory!\n";
178 std::exit(EXIT_FAILURE);
179 }
180 *hits = 0;
181
182 thread_local std::mt19937_64 engine(std::random_device{}());
183 std::uniform_real_distribution<double> dist(0.0, 1.0);
184
185 int batch;
186 int loopEnd;
187
188 #ifdef USE_AVX
189 batch = 4;
190 alignas(32) double randX[batch], randY[batch];
191 loopEnd = numberOfTrials - (numberOfTrials % batch);
192
193 for (int i = 0; i < loopEnd; i+= batch) {
194 for (int j = 0; j < batch; ++j) {
195 randX[j] = dist(engine);
196 randY[j] = dist(engine);
197 }
198 __m256d dartX = _mm256_load_pd(randX);
199 __m256d dartY = _mm256_load_pd(randY);
200 *hits += countInsideCircle_AVX(dartX, dartY);
201 }
202 #elif defined(USE_NEON)
203 batch = 2;
204 alignas(16) double randX[batch], randY[batch];
205 loopEnd = numberOfTrials - (numberOfTrials % batch);
206
207 for (int i = 0; i < loopEnd; i += batch) {
208 for (int j = 0; j < batch; ++j) {
209 randX[j] = dist(engine);
210 randY[j] = dist(engine);
211 }
212 float64x2_t dartX = vld1q_f64(randX);
213 float64x2_t dartY = vld1q_f64(randY);
214 *hits += countInsideCircle_NEON(dartX, dartY);
215 }
216 #endif
217
218 for (int i = loopEnd; i < numberOfTrials; ++i) {
219 double dartX = dist(engine);
220 double dartY = dist(engine);
221 if (isInsideCircle(dartX, dartY)) ++(*hits);
222 }
223 return hits;
